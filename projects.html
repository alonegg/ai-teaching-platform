核心代码示例

sentiment_analyzer.py 
import tkinter as tk
from tkinter import ttk, filedialog, messagebox
from textblob import TextBlob
import matplotlib.pyplot as plt
import pandas as pd

class SentimentAnalyzer:
    def __init__(self, root):
        self.root = root
        self.root.title("情感分析工具")
        self.root.geometry("800x600")
        
        # 创建主界面
        self.create_widgets()
        
    def analyze_sentiment(self, text):
        """分析文本情感"""
        blob = TextBlob(text)
        polarity = blob.sentiment.polarity
        subjectivity = blob.sentiment.subjectivity
        
        if polarity > 0.1:
            sentiment = "积极"
        elif polarity < -0.1:
            sentiment = "消极"
        else:
            sentiment = "中性"
            
        return {
            'sentiment': sentiment,
            'polarity': polarity,
            'subjectivity': subjectivity
        }
    
    def process_files(self):
        """批量处理文件"""
        files = filedialog.askopenfilenames(
            title="选择文本文件",
            filetypes=[("文本文件", "*.txt"), ("所有文件", "*.*")]
        )
        
        if files:
            results = []
            for file_path in files:
                with open(file_path, 'r', encoding='utf-8') as f:
                    content = f.read()
                    result = self.analyze_sentiment(content)
                    result['file'] = file_path.split('/')[-1]
                    results.append(result)
            
            self.display_results(results)
            self.create_visualization(results)
    
    def create_visualization(self, results):
        """创建可视化图表"""
        df = pd.DataFrame(results)
        
        plt.figure(figsize=(12, 8))
        
        # 情感分布饼图
        plt.subplot(2, 2, 1)
        sentiment_counts = df['sentiment'].value_counts()
        plt.pie(sentiment_counts.values, labels=sentiment_counts.index, autopct='%1.1f%%')
        plt.title('情感分布')
        
        # 情感极性分布
        plt.subplot(2, 2, 2)
        plt.hist(df['polarity'], bins=20, alpha=0.7)
        plt.xlabel('情感极性')
        plt.ylabel('频次')
        plt.title('情感极性分布')
        
        # 文件情感对比
        plt.subplot(2, 1, 2)
        plt.bar(range(len(df)), df['polarity'])
        plt.xlabel('文件')
        plt.ylabel('情感极性')
        plt.title('各文件情感对比')
        plt.xticks(range(len(df)), df['file'], rotation=45)
        
        plt.tight_layout()
        plt.show()

使用指南

快速开始

  1. 下载并解压项目文件
  2. 安装Python 3.8或更高版本
  3. 运行 pip install -r requirements.txt 安装依赖
  4. 执行 python sentiment_analyzer.py 启动程序
  5. 选择文本文件进行分析
  6. 查看分析结果和可视化图表
  7. 导出结果到Excel或图片格式
